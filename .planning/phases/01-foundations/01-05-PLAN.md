Create `.github/actions/build-zig-binding/action.yml`:

    name: Build Zig binding
    description: >-
      Installs Zig 0.16.0 via mlugg/setup-zig, fetches napi-zig, builds shoki.node
      for the requested target triple, and places it into the matching platform
      binding package directory.

    inputs:
      target:
        description: "Zig target triple: aarch64-macos or x86_64-macos"
        required: true
      binding-package-dir:
        description: "Path to the @shoki/binding-<os>-<arch> package directory"
        required: true
      run-tests:
        description: "Run zig build test after the build"
        required: false
        default: "false"

    runs:
      using: composite
      steps:
        - name: Setup Zig 0.16.0
          uses: mlugg/setup-zig@v2
          with:
            version: 0.16.0

        - name: Print Zig version
          shell: bash
          run: zig version

        - name: Fetch napi-zig dependency
          shell: bash
          working-directory: zig
          run: |
            set -euo pipefail
            # Extract the pinned napi-zig SHA URL from build.zig.zon and run `zig fetch --save`
            # to populate the .hash field reproducibly. If the zon already has a concrete hash,
            # this is a no-op.
            NAPI_URL=$(grep -oE 'git\+https://github.com/yuku-toolchain/napi-zig\.git#[0-9a-f]+' build.zig.zon || true)
            if [[ -z "$NAPI_URL" ]]; then
                echo "error: could not find napi-zig dependency URL in build.zig.zon" >&2
                exit 1
            fi
            zig fetch --save=napi_zig "$NAPI_URL"

        - name: Zig unit tests
          if: inputs.run-tests == 'true'
          shell: bash
          working-directory: zig
          run: zig build test -Dtarget=${{ inputs.target }}

        - name: Build shared library
          shell: bash
          working-directory: zig
          run: zig build -Doptimize=ReleaseSafe -Dtarget=${{ inputs.target }}

        - name: Locate and rename output
          shell: bash
          working-directory: zig
          run: |
            set -euo pipefail
            # The produced shared library name depends on Zig's default (libshoki.dylib on macOS).
            # Rename to shoki.node for the N-API loader.
            OUT=$(find zig-out/lib -name "libshoki.dylib" -o -name "shoki.dylib" | head -n1)
            if [[ -z "$OUT" ]]; then
                echo "error: built shared library not found under zig-out/lib" >&2
                ls -la zig-out/lib || true
                exit 1
            fi
            TARGET_DIR="$GITHUB_WORKSPACE/${{ inputs.binding-package-dir }}"
            mkdir -p "$TARGET_DIR"
            cp "$OUT" "$TARGET_DIR/shoki.node"
            ls -la "$TARGET_DIR/shoki.node"

        - name: Verify .node loads
          shell: bash
          working-directory: ${{ inputs.binding-package-dir }}
          run: |
            # Only attempts a dlopen test when host arch matches the target — an arm64 runner
            # cannot require() an x86_64 .node. Skip quietly when mismatched; the matching
            # runner validates the other triple.
            HOST_ARCH=$(uname -m)
            case "${{ inputs.target }}" in
                aarch64-*) WANT_ARCH=arm64 ;;
                x86_64-*)  WANT_ARCH=x86_64 ;;
                *)         WANT_ARCH=unknown ;;
            esac
            if [[ "$HOST_ARCH" != "$WANT_ARCH" ]]; then
                echo "host=$HOST_ARCH target=${{ inputs.target }}; skipping dlopen smoke test"
                exit 0
            fi
            node -e "const b=require('./shoki.node'); if(b.ping()!=='pong'){process.exit(1)}; console.log('ping round-trip: ok')"

Create `.github/actions/build-helper-app/action.yml`:

    name: Build and sign ShokiRunner.app
    description: >-
      Builds ShokiRunner.app via SPM, imports Developer ID cert from secrets,
      codesigns with hardened runtime + entitlements, and notarizes via notarytool.
      The resulting signed bundle is copied into the provided binding package dir
      as helper/ShokiRunner.app.

    inputs:
      binding-package-dir:
        description: "Destination binding package dir (helper/ShokiRunner.app copied inside)"
        required: true
      apple-cert-p12-base64:
        description: "Base64-encoded Developer ID Application cert (.p12)"
        required: true
      apple-cert-password:
        description: "Password for the p12 cert"
        required: true
      apple-developer-id-app:
        description: "Common name: 'Developer ID Application: Name (TEAMID)'"
        required: true
      apple-id:
        description: "Apple ID email for notarytool"
        required: true
      apple-team-id:
        description: "10-char team id"
        required: true
      apple-app-specific-password:
        description: "App-specific password for notarytool"
        required: true

    runs:
      using: composite
      steps:
        - name: Import Developer ID cert into keychain
          uses: apple-actions/import-codesign-certs@v3
          with:
            p12-file-base64: ${{ inputs.apple-cert-p12-base64 }}
            p12-password: ${{ inputs.apple-cert-password }}

        - name: Run Swift tests
          shell: bash
          working-directory: helper
          run: swift test

        - name: Build app bundle
          shell: bash
          working-directory: helper
          run: scripts/build-app-bundle.sh --configuration release

        - name: Sign and notarize
          shell: bash
          working-directory: helper
          env:
            APPLE_DEVELOPER_ID_APP: ${{ inputs.apple-developer-id-app }}
            APPLE_ID: ${{ inputs.apple-id }}
            APPLE_TEAM_ID: ${{ inputs.apple-team-id }}
            APPLE_APP_SPECIFIC_PASSWORD: ${{ inputs.apple-app-specific-password }}
          run: scripts/sign-and-notarize.sh ./ShokiRunner.app

        - name: Verify Developer ID signature present
          shell: bash
          working-directory: helper
          run: |
            set -euo pipefail
            OUT=$(codesign -dvvv ./ShokiRunner.app 2>&1)
            echo "$OUT"
            if ! echo "$OUT" | grep -q "Authority=Developer ID Application"; then
                echo "error: Developer ID Application authority not found in codesign output" >&2
                exit 1
            fi
            if ! echo "$OUT" | grep -q "Authority=Developer ID Certification Authority"; then
                echo "error: Developer ID chain not found in codesign output" >&2
                exit 1
            fi
            echo "codesign verification: Developer ID chain confirmed"

        - name: Copy signed bundle into binding package
          shell: bash
          run: |
            set -euo pipefail
            DEST="$GITHUB_WORKSPACE/${{ inputs.binding-package-dir }}/helper"
            mkdir -p "$DEST"
            # Preserve the .app bundle structure with ditto (cp -R can mangle xattrs)
            /usr/bin/ditto helper/ShokiRunner.app "$DEST/ShokiRunner.app"
            ls -la "$DEST"
  </action>

Create `.github/workflows/release.yml`:

    name: Release

    on:
      push:
        tags:
          - "v*.*.*"

    permissions:
      contents: write   # create GitHub Release
      id-token: write   # npm OIDC trusted publishing
      attestations: write

    concurrency:
      group: release-${{ github.ref }}
      cancel-in-progress: false

    jobs:
      build-binding:
        name: Build binding ${{ matrix.target }}
        runs-on: macos-14
        strategy:
          fail-fast: true
          matrix:
            include:
              - target: aarch64-macos
                binding-dir: packages/binding-darwin-arm64
              - target: x86_64-macos
                binding-dir: packages/binding-darwin-x64
        steps:
          - uses: actions/checkout@v4
          - uses: pnpm/action-setup@v4
            with:
              version: 10
          - uses: actions/setup-node@v4
            with:
              node-version: "24"
              cache: pnpm
          - run: pnpm install --frozen-lockfile

          - name: Build Zig binding
            uses: ./.github/actions/build-zig-binding
            with:
              target: ${{ matrix.target }}
              binding-package-dir: ${{ matrix.binding-dir }}
              run-tests: "true"

          - name: Build + sign + notarize ShokiRunner.app
            uses: ./.github/actions/build-helper-app
            with:
              binding-package-dir: ${{ matrix.binding-dir }}
              apple-cert-p12-base64: ${{ secrets.APPLE_DEVELOPER_ID_CERT_P12 }}
              apple-cert-password: ${{ secrets.APPLE_DEVELOPER_ID_CERT_PASSWORD }}
              apple-developer-id-app: ${{ secrets.APPLE_DEVELOPER_ID_APP }}
              apple-id: ${{ secrets.APPLE_ID }}
              apple-team-id: ${{ secrets.APPLE_TEAM_ID }}
              apple-app-specific-password: ${{ secrets.APPLE_APP_SPECIFIC_PASSWORD }}

          - name: Upload binding artifact
            uses: actions/upload-artifact@v4
            with:
              name: binding-${{ matrix.target }}
              path: |
                ${{ matrix.binding-dir }}/shoki.node
                ${{ matrix.binding-dir }}/helper/ShokiRunner.app
              if-no-files-found: error
              retention-days: 7

      publish:
        name: Publish npm packages via OIDC
        needs: [build-binding]
        runs-on: macos-14
        steps:
          - uses: actions/checkout@v4
          - uses: pnpm/action-setup@v4
            with:
              version: 10
          - uses: actions/setup-node@v4
            with:
              node-version: "24"
              registry-url: "https://registry.npmjs.org"
              cache: pnpm

          - name: Download arm64 binding artifact
            uses: actions/download-artifact@v4
            with:
              name: binding-aarch64-macos
              path: packages/binding-darwin-arm64

          - name: Download x64 binding artifact
            uses: actions/download-artifact@v4
            with:
              name: binding-x86_64-macos
              path: packages/binding-darwin-x64

          - run: pnpm install --frozen-lockfile
          - name: Build SDK
            run: pnpm --filter @shoki/sdk build

          - name: Verify signed helper present in both binding packages
            shell: bash
            run: |
              set -euo pipefail
              for dir in packages/binding-darwin-arm64 packages/binding-darwin-x64; do
                  if [[ ! -f "$dir/shoki.node" ]]; then
                      echo "error: $dir/shoki.node missing" >&2; exit 1
                  fi
                  if [[ ! -d "$dir/helper/ShokiRunner.app" ]]; then
                      echo "error: $dir/helper/ShokiRunner.app missing" >&2; exit 1
                  fi
                  # Developer ID check only runs on macos-14 which has codesign
                  OUT=$(codesign -dvvv "$dir/helper/ShokiRunner.app" 2>&1)
                  echo "$OUT"
                  if ! echo "$OUT" | grep -q "Authority=Developer ID Application"; then
                      echo "error: $dir helper app is not Developer ID-signed" >&2; exit 1
                  fi
              done

          - name: Set version from tag
            shell: bash
            run: |
              set -euo pipefail
              VERSION="${GITHUB_REF_NAME#v}"
              echo "Releasing version: $VERSION"
              pnpm version "$VERSION" --no-git-tag-version --workspaces

          - name: Publish all packages (OIDC trusted publishing, no NPM_TOKEN)
            # id-token: write is provided by the workflow-level permissions block above.
            # Each package's npmjs.com settings must have a trusted publisher configured
            # that matches org=<this repo's org>, workflow=release.yml, environment=(none).
            # See docs/release-setup.md.
            shell: bash
            run: pnpm publish -r --no-git-checks --access public

          - name: Create GitHub Release
            uses: softprops/action-gh-release@v2
            with:
              generate_release_notes: true
              files: |
                packages/binding-darwin-arm64/shoki.node
                packages/binding-darwin-x64/shoki.node

    # Release setup

    Operational checklist for Shoki's release pipeline. Complete every step BEFORE pushing the first `v*.*.*` tag.

    ## 1. Reserve the `@shoki` npm scope

    - Create the `shoki` organization on https://www.npmjs.com/.
    - Add yourself (and backup owner) as Admin.
    - Keep 2FA enforced at the organization level.

    ## 2. Configure npm OIDC trusted publishing — three packages

    We publish WITHOUT `NPM_TOKEN`. Every package must have a Trusted Publisher configured on npmjs.com.

    For each of:
    - `@shoki/sdk`
    - `@shoki/binding-darwin-arm64`
    - `@shoki/binding-darwin-x64`

    Do the first publish manually (one-time bootstrap) so the package exists, then:

    1. Navigate to https://www.npmjs.com/package/<pkg>/access
    2. Scroll to **Trusted Publishers**
    3. Click **+ Add Publisher**
    4. Fill in:
       - Publisher: GitHub Actions
       - Organization / User: `shoki` (or your fork's org)
       - Repository: `shoki`
       - Workflow filename: `release.yml`
       - Environment name: (leave blank unless you use GH environments)
    5. Save

    Reference: https://docs.npmjs.com/trusted-publishers

    Once configured, the workflow authenticates via the `id-token: write` permission in `release.yml` — no secret required.

    ## 3. Apple Developer Program setup

    ### 3.1 Developer ID Application cert

    Needed to codesign `ShokiRunner.app` (helper bundle).

    1. Log in at https://developer.apple.com/account
    2. Navigate to **Certificates, Identifiers & Profiles** → **Certificates**
    3. Click **+**, pick **Developer ID Application**, follow the CSR flow
    4. Download the `.cer` and import into Keychain Access (usually it's already there from the CSR step)
    5. In Keychain Access, right-click the cert and **Export…** as a `.p12` file — set a password (you'll store this as a secret)
    6. Base64-encode the p12:

           base64 -i /path/to/ShokiDeveloperIDApp.p12 | pbcopy

       (the output in your clipboard becomes the `APPLE_DEVELOPER_ID_CERT_P12` secret)

    ### 3.2 notarytool credentials

    Needed to notarize `ShokiRunner.app`.

    1. Go to https://appleid.apple.com → **Sign-In and Security** → **App-Specific Passwords**
    2. Generate a password labelled `shoki-notarytool`
    3. Record:
       - `APPLE_ID` = the email of the Apple ID holding your developer program seat
       - `APPLE_TEAM_ID` = the 10-char team id shown in developer.apple.com → Membership
       - `APPLE_APP_SPECIFIC_PASSWORD` = the password generated above

    ### 3.3 Developer ID common name

    Run `security find-identity -v -p codesigning` on a Mac with the cert installed. Copy the full common name (it looks like `Developer ID Application: Your Name (ABCDE12345)`). This is `APPLE_DEVELOPER_ID_APP`.

    ## 4. GitHub repository secrets

    Go to **Repository → Settings → Secrets and variables → Actions → Secrets** and add:

    | Secret name | Value |
    |---|---|
    | `APPLE_DEVELOPER_ID_CERT_P12` | base64 output from step 3.1.6 |
    | `APPLE_DEVELOPER_ID_CERT_PASSWORD` | p12 export password from step 3.1.5 |
    | `APPLE_DEVELOPER_ID_APP` | common name from step 3.3 |
    | `APPLE_ID` | from step 3.2.3 |
    | `APPLE_TEAM_ID` | from step 3.2.3 |
    | `APPLE_APP_SPECIFIC_PASSWORD` | from step 3.2.3 |

    **Do NOT create an `NPM_TOKEN` secret.** The pipeline uses OIDC trusted publishing; adding a token defeats the security model.

    ## 5. First release (bootstrap)

    The first time you publish, OIDC requires the package to exist on npm. You have two options:

    **Option A — manual bootstrap:** Temporarily add an `NPM_TOKEN` secret, run `pnpm publish -r --access public` once locally (or from a one-off workflow), then delete the token and configure trusted publishers (step 2).

    **Option B — scoped token bootstrap:** Use an npm token scoped to "Automation: publish" for just the first release, then swap to OIDC.

    After the first release, subsequent tag pushes will publish via OIDC only.

    ## 6. Triggering a release

    1. Commit your changes to `main`
    2. Tag the release:

           git tag v0.1.0
           git push origin v0.1.0

    3. Watch `.github/workflows/release.yml` in the GitHub Actions tab
    4. On success you will see:
       - Zig binding built for darwin-arm64 + darwin-x64
       - `ShokiRunner.app` codesigned + notarized (codesign -dvvv grep passes)
       - Three packages published to npm (verify on npmjs.com)
       - A GitHub Release created with generated notes

    ## 7. Verifying a release

    On a fresh machine:

        npm install @shoki/sdk
        node -e "console.log(require('@shoki/sdk').ping())"
        # expect: pong
        codesign -dvvv node_modules/@shoki/binding-darwin-arm64/helper/ShokiRunner.app 2>&1 | grep Authority
        # expect: Authority=Developer ID Application: ...

    If the codesign verify fails, the release pipeline had a regression — rollback the npm version immediately and investigate before re-releasing.

    ## 8. Rotating secrets

    - **Developer ID cert:** renew before expiry (5 years). When renewing, export a new p12 and update `APPLE_DEVELOPER_ID_CERT_P12` + password. Do NOT remove the old one until the new cert has shipped a successful release (signing chain overlap).
    - **App-specific password:** rotate yearly or when anyone with access leaves. Regenerate at appleid.apple.com; update `APPLE_APP_SPECIFIC_PASSWORD`.
    - **npm trusted publisher:** if you rename the repo or workflow file, update the trusted publisher entry on npmjs.com.

    ## 9. Pitfall reminders (see .planning/research/PITFALLS.md)

    - **Pitfall 4:** TCC grants are per-code-signature. Rotating the Developer ID cert invalidates grants until users re-grant Accessibility to the new ShokiRunner.app. Call this out in release notes for any cert-change release.
    - **Pitfall 12:** Never rely on `postinstall` scripts. Binaries ship in the platform packages; corporate `--ignore-scripts` CI works out of the box.

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-20 | Spoofing | npm package hijack via token leak | mitigate | OIDC trusted publishing — no long-lived `NPM_TOKEN` to leak. Grep in verify step confirms no NPM_TOKEN under .github/. |
