async function warnOnLegacyStateDir() {

   import os from "node:os";
   import path from "node:path";

   function warnOnLegacyStateDir(): void {
     const legacy = [".shoki", ".dicta", ".munadi"];
     const home = os.homedir();
     const found = legacy.filter((d) => existsSync(path.join(home, d)));

     }
   }
