       ```ts
       import type { ShokiEvent } from './screen-reader.js';

       export class LogStore {
         private events: ShokiEvent[] = [];
         private target = new EventTarget();
         private _droppedCount = 0n;

         push(event: ShokiEvent): void {
           this.events.push(event);
           this.target.dispatchEvent(new CustomEvent('event', { detail: event }));
         }

         pushMany(events: ShokiEvent[]): void {
           for (const e of events) this.push(e);
         }

         setDroppedCount(n: bigint): void {
           this._droppedCount = n;
         }

         droppedCount(): bigint {
           return this._droppedCount;
         }

         getAll(): ShokiEvent[] {
           return this.events.slice();
         }

         lastPhrase(): string | undefined {
           return this.events[this.events.length - 1]?.phrase;
         }

         clear(): void {
           this.events.length = 0;
           // Intentionally preserve droppedCount — user can observe total loss across clears
         }

         subscribe(fn: (event: ShokiEvent) => void): () => void {
           const handler = (e: Event) => fn((e as CustomEvent<ShokiEvent>).detail);
           this.target.addEventListener('event', handler);
           return () => this.target.removeEventListener('event', handler);
         }

         /** Internal: the last-N cursor used by awaitStableLog */
         now(): number {
           return this.events.length;
         }
       }
       ```

    2. Create `packages/sdk/src/listen.ts`:
       ```ts
       import type { ShokiEvent } from './screen-reader.js';
       import { LogStore } from './handle-internals.js';

       /** Async generator yielding events as they arrive. Consumes the full history first,
        * then awaits new events via a microtask-resolved queue. Safe to cancel via `.return()`. */
       export async function* listenImpl(store: LogStore, signal?: AbortSignal): AsyncGenerator<ShokiEvent> {
         const queue: ShokiEvent[] = store.getAll();
         let waiters: Array<(e: ShokiEvent | null) => void> = [];
         let closed = false;

         const unsubscribe = store.subscribe((ev) => {
           if (closed) return;
           if (waiters.length > 0) {
             waiters.shift()!(ev);
           } else {
             queue.push(ev);
           }
         });

         const abortHandler = () => {
           closed = true;
           while (waiters.length) waiters.shift()!(null);
         };
         signal?.addEventListener('abort', abortHandler, { once: true });

         try {
           while (!closed) {
             if (queue.length > 0) {
               yield queue.shift()!;
               continue;
             }
             const next = await new Promise<ShokiEvent | null>((resolve) => waiters.push(resolve));
             if (next === null) break;
             yield next;
           }
         } finally {
           unsubscribe();
           signal?.removeEventListener('abort', abortHandler);
         }
       }
       ```

       ```ts
       export class LogStore {
         // ... existing members ...

         awaitStable(opts: AwaitStableLogOptions): Promise<ShokiEvent[]> {
           const { quietMs, signal } = opts;
           if (!Number.isFinite(quietMs) || quietMs < 0) {
             throw new TypeError(`awaitStableLog: quietMs must be a non-negative finite number, got ${quietMs}`);
           }
           return new Promise<ShokiEvent[]>((resolve, reject) => {
             let timer: ReturnType<typeof setTimeout> | undefined;
             const unsubscribe = this.subscribe(() => {
               if (timer !== undefined) clearTimeout(timer);
               timer = setTimeout(settle, quietMs);
             });

             const abortHandler = () => {
               if (timer !== undefined) clearTimeout(timer);
               unsubscribe();
               signal?.removeEventListener('abort', abortHandler);
               reject(new DOMException('Aborted', 'AbortError'));
             };
             if (signal) {
               if (signal.aborted) return abortHandler();
               signal.addEventListener('abort', abortHandler, { once: true });
             }

             const settle = () => {
               unsubscribe();
               signal?.removeEventListener('abort', abortHandler);
               resolve(this.getAll());
             };

             timer = setTimeout(settle, quietMs);
           });
         }
       }
       ```
