           ax: ?*ax_mod.AxNotifications = null,

               clock: lifecycle_mod.Clock,
               spawner: applescript_mod.ChildProcessSpawner,
               xpc_backend: ax_mod.XpcBackend,

               const self = try allocator.create(VoiceOverDriver);
               self.* = .{
                   .allocator = allocator,
                   .lifecycle = undefined,  // initialized in initImpl when we have InitOptions
                   .ring = try rb_mod.RingBuffer.init(allocator, 10_000),  // capacity refined in initImpl
                   .runner = runner,
                   .clock = clock,
                   .spawner = spawner,
                   .xpc_backend = xpc_backend,
               };
               return self;

               return .{
                   .init = initImpl,
                   .start = startImpl,
                   .stop = stopImpl,
                   .drain = drainImpl,
                   .reset = resetImpl,
                   .deinit = deinitImpl,
                   .name = "voiceover",
                   .platform = .darwin,
               };
           }

           fn initImpl(ctx: *anyopaque, opts: opts_mod.InitOptions) anyerror!void {
               const self: *VoiceOverDriver = @ptrCast(@alignCast(ctx));
               self.init_opts = opts;
               // Resize ring to user-requested capacity
               self.ring.deinit();
               self.ring = try rb_mod.RingBuffer.init(self.allocator, opts.log_buffer_size);
               self.lifecycle = try lifecycle_mod.Lifecycle.init(
                   self.allocator, self.runner, self.clock, self.spawner, opts,
               );
               lifecycle_mod.installExitHooks(&self.lifecycle);
           }

           fn startImpl(ctx: *anyopaque) anyerror!void {
               const self: *VoiceOverDriver = @ptrCast(@alignCast(ctx));
               try self.lifecycle.startHandle();  // boots VO, snapshots plist, opens OsascriptShell

               // Start AppleScript poll loop
               const shell_ptr = self.lifecycle.getShell() orelse return error.VoiceOverShellMissing;
               const poll = try self.allocator.create(applescript_mod.PollLoop);
               poll.* = applescript_mod.PollLoop.init(
                   self.allocator, shell_ptr, &self.ring, self.clock, .{},
               );
               try poll.start();
               self.poll_loop = poll;

               // Start AX observer
               const ax = try self.allocator.create(ax_mod.AxNotifications);
               ax.* = ax_mod.AxNotifications.init(self.allocator, &self.ring, self.xpc_backend);
               // Obtain VO PID via runner (pgrep -x VoiceOver) — do here, not in ax module
               const vopid = try resolveVoiceOverPid(self.allocator, self.runner);
               try ax.start(vopid);
               self.ax = ax;
           }

           fn stopImpl(ctx: *anyopaque) anyerror!void {
               const self: *VoiceOverDriver = @ptrCast(@alignCast(ctx));
               if (self.ax) |ax| { ax.stop(); ax.deinit(); self.allocator.destroy(ax); self.ax = null; }
               if (self.poll_loop) |p| { p.stop(); p.deinit(); self.allocator.destroy(p); self.poll_loop = null; }
               try self.lifecycle.stopHandle();
           }

           fn drainImpl(ctx: *anyopaque, out: *rb_mod.RingBuffer) anyerror!usize {
               const self: *VoiceOverDriver = @ptrCast(@alignCast(ctx));
               // Transfer entries from our ring to `out`; Phase 1's napi.zig then encodes `out` via wire.encode.
               // If RingBuffer has a `moveInto(other)` API, use it. Else iterate drain().
               return try self.ring.drainInto(out);  // inspect ring_buffer.zig for the exact method name
           }

           fn resetImpl(ctx: *anyopaque) anyerror!void {
               const self: *VoiceOverDriver = @ptrCast(@alignCast(ctx));
               self.ring.clear();
               // Best-effort VO cursor reset
               if (self.lifecycle.getShell()) |shell| {
                   _ = shell.sendAndReceive(
                       "tell application \"VoiceOver\" to tell vo cursor to move to first item of window 1",
                       2000,
                   ) catch {};
               }
           }

           fn deinitImpl(ctx: *anyopaque) void {
               const self: *VoiceOverDriver = @ptrCast(@alignCast(ctx));
               lifecycle_mod.uninstallExitHooks();
               self.lifecycle.deinit();
               self.ring.deinit();
               self.allocator.destroy(self);
           }

           fn resolveVoiceOverPid(allocator, runner: defaults_mod.SubprocessRunner) !i32 {
               // pgrep -x VoiceOver -> parse stdout as i32
               var result = try runner.run(runner.ctx, allocator, &[_][]const u8{ "/usr/bin/pgrep", "-x", "VoiceOver" });
               defer result.deinit(allocator);
               if (result.exit_code != 0) return error.VOPidNotFound;
               const trimmed = std.mem.trim(u8, result.stdout, " \n\r\t");
               return try std.fmt.parseInt(i32, trimmed, 10);
           }
       };
