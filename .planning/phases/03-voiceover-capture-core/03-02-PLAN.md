<applescript_payload>
Poll command (written to osascript stdin, \n-terminated). Wrapped in transaction for atomicity (CONTEXT.md D-02):

```
tell application "VoiceOver"
    with transaction
        return content of last phrase
    end transaction
end tell
```

Expected osascript invocation: `osascript -s s -i` where:
- `-s s` = "stay-open" mode: exit on stdin EOF, not on end-of-script. Each script is terminated by an empty line or by `on error` boundary.
- `-i` = interactive mode (reads from stdin)

Actual osascript argv: `["/usr/bin/osascript", "-s", "s", "-i", "-"]` (the trailing `-` forces stdin script reading in some macOS versions; test on 14/15/26)

Result format:
- Successful read: one line of text on stdout, followed by a shell prompt-like boundary
- Error: osascript emits on stderr; stdout may be empty

Separator strategy: send a sentinel `log "__SHOKI_SEP__"` after each real script so the reader knows when one reply ends. Parse stdout lines until `__SHOKI_SEP__` appears.
</applescript_payload>

           allocator: std.mem.Allocator,
           child: *ChildProcess,
           spawner: ChildProcessSpawner,

           pub fn init(allocator, spawner: ChildProcessSpawner) !OsascriptShell {
               const child = try spawner.spawn(spawner.ctx, allocator, &[_][]const u8{
                   "/usr/bin/osascript", "-s", "s", "-i", "-",
               });
               return .{ .allocator = allocator, .child = child, .spawner = spawner };
           }
           pub fn sendAndReceive(self: *OsascriptShell, script: []const u8, timeout_ms: u32) ![]u8 {
               const wrapped = try wrapInTransaction(self.allocator, script);
               defer self.allocator.free(wrapped);
               try self.child.write_stdin(self.child, wrapped);
               try self.child.write_stdin(self.child, "\n");
               // Read lines until the sentinel; return the phrase body (lines before sentinel).
               // On stall (read_stdout_line returns error.Timeout), return error.OsascriptStall.
               // On the first line being "osascript error ...", return error.OsascriptError.
           }
           pub fn deinit(self: *OsascriptShell) void { ... }
           pub fn respawn(self: *OsascriptShell) !void {
               self.child.kill(self.child);
               self.child.wait(self.child);
               self.child.deinit_ctx(self.child);
               self.child = try self.spawner.spawn(self.spawner.ctx, self.allocator, &[_][]const u8{
                   "/usr/bin/osascript", "-s", "s", "-i", "-",
               });
           }
       };

       ```zig
       pub const PollLoopOptions = struct {
           interval_ms: u32 = 50,
           stall_timeout_ms: u32 = 500,
           max_consecutive_stalls: u32 = 10,
       };

       pub const PollLoop = struct {
           allocator: std.mem.Allocator,
           shell: *OsascriptShell,
           ring: *RingBuffer, // from core
           clock: Clock,
           options: PollLoopOptions,
           thread: ?std.Thread = null,
           stop_flag: std.atomic.Value(bool) = std.atomic.Value(bool).init(false),
           degraded_flag: std.atomic.Value(bool) = std.atomic.Value(bool).init(false),
           last_phrase: ?[]u8 = null,             // owned; freed on dedup replace
           consecutive_stalls: u32 = 0,

           pub fn init(allocator, shell, ring, clock, options) PollLoop { ... }
           pub fn start(self: *PollLoop) !void {
               self.thread = try std.Thread.spawn(.{}, runLoop, .{self});
           }
           pub fn stop(self: *PollLoop) void {
               self.stop_flag.store(true, .release);
               if (self.thread) |t| { t.join(); self.thread = null; }
           }
           pub fn isDegraded(self: *const PollLoop) bool {
               return self.degraded_flag.load(.acquire);
           }
           pub fn deinit(self: *PollLoop) void {
               if (self.last_phrase) |p| self.allocator.free(p);
           }

           fn runLoop(self: *PollLoop) void {
               while (!self.stop_flag.load(.acquire)) {
                   const result = self.shell.sendAndReceive("return content of last phrase", self.options.stall_timeout_ms);
                   // handle result per behavior cases
                   self.clock.sleep_ms(self.clock.ctx, self.options.interval_ms);
               }
           }
       };
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Zig poll thread ↔ `osascript` subprocess | Scripts are hardcoded strings — no user input ever reaches this boundary |
| `osascript` ↔ VoiceOver via Apple Events | Mediated by the `with transaction` wrap |
| Poll thread ↔ ring buffer | Lock-free; the ring's push is thread-safe per Phase 1 contract |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-10 | Tampering | AppleScript body via stdin | accept | Scripts are compile-time constants; no user input flows to osascript |
| T-03-11 | Denial of service | hung osascript child | mitigate | stall_timeout_ms=500 + respawn; max_consecutive_stalls=10 → degraded exit |
| T-03-12 | Denial of service | PollLoop thread doesn't exit on stop | mitigate | stop_flag is atomic release; thread checks on every iteration; join has upper bound = interval_ms + stall_timeout_ms |
| T-03-13 | Information disclosure | phrase content (user-spoken text) in logs | accept | Phrase content is the test observable; user knowingly captures it |
| T-03-14 | Spoofing | a malicious process responds as VoiceOver to our AppleScript | accept | `tell application "VoiceOver"` binds by bundle identifier; out of scope for this plan, Plan 03 will reconcile the VO process identity at start |
</threat_model>
