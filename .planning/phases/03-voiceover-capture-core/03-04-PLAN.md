<ax_api_reference>
<!-- Apple AXObserver API surface -->

Framework: `ApplicationServices` (includes HIServices which contains AXUIElement + AXObserver headers)

Key calls (Swift CoreFoundation bridge):
```swift
import ApplicationServices

// Create the system-wide AXUIElement (represents all accessible elements on the system)
let systemElement: AXUIElement = AXUIElementCreateSystemWide()

// Create an observer for a PID (we pass 0 / or the VO process PID once reconciled — see below)
var observer: AXObserver?
let err: AXError = AXObserverCreate(pid_t, callback, &observer)

// Add the announcement-requested notification
// Third param is the AXUIElement we want to observe; systemElement is correct for VO announcements
AXObserverAddNotification(observer!, systemElement, kAXAnnouncementRequestedNotification as CFString, /*refCon*/ nil)

// Attach the observer's run-loop source to the CURRENT run loop — must be a thread with a running CFRunLoop
CFRunLoopAddSource(
  CFRunLoopGetCurrent(),
  AXObserverGetRunLoopSource(observer!),
  .defaultMode
)
```

Callback shape (C function pointer):
```swift
func axCallback(_ observer: AXObserver,
                _ element: AXUIElement,
                _ notification: CFString,
                _ userInfo: CFDictionary?,
                _ refcon: UnsafeMutableRawPointer?) {
    // userInfo contains: kAXAnnouncementKey ("announcement" text), kAXUIElementTitleKey (optional title)
    // refcon is the context pointer we passed in (we'll put a pointer to our session here)
}
```

userInfo keys (strings documented in AXConstants.h — hardcode as Swift literals):
- `"AXAnnouncement"` → the announcement text (String)
- `"AXPriority"` → priority level (Int; optional)
- `"AXUIElementTitle"` → source element title (String; optional — use as `name` in our Entry)

PID to observe: the VoiceOver process PID. Plan 03 booted VO; for this plan we accept a `pid_t` parameter from the Zig side (via XPC) — the Zig side will call `pgrep -x VoiceOver` to get the PID, then pass it to the helper's startAXObserver.

TCC note: AX observation requires Accessibility TCC grant. The helper holds this grant (Phase 1 decision) — granted to ShokiRunner.app, not to `node` or the Zig binary. The user granted this via `shoki doctor --fix` (Phase 2).

<xpc_callback_strategy>
<!-- How the helper forwards events back to Zig -->

Two options for the helper → Zig callback:

Option A (used in this plan): **reverse XPC** — Zig exposes an NSXPCListener; helper's `startAXObserver` takes an NSXPCListenerEndpoint as an argument, the helper opens a connection to that endpoint and calls a `ShokiClientProtocol.receiveAXEvent(phrase, ts, roleOpt, nameOpt)` on it.

Option B: **exportedObject pattern** — the initial NSXPCConnection from Zig → helper has `remoteObjectInterface` set to ShokiRunnerProtocol AND `exportedInterface` set to ShokiClientProtocol; helper calls back on the same connection.

Option B is simpler and more idiomatic for NSXPCConnection. Use Option B.

Swift side (ShokiClientProtocol lives in the Protocol module so both sides see it):
```swift
@objc public protocol ShokiClientProtocol {
    func receiveAXEvent(phrase: String, tsNanos: UInt64, role: String?, name: String?)
}
```

Helper service:
```swift
public func startAXObserver(vopid: pid_t, reply: @escaping (Error?) -> Void) {
    // Caller's connection was captured earlier via NSXPCListenerDelegate (see Task 1).
    // Retrieve the remoteObjectProxy of ShokiClientProtocol on that connection and
    // store it in the AXObserverSession so the AX callback can invoke it.
}
```

Zig side: the initial NSXPCConnection from Zig sets `exportedInterface` to ShokiClientProtocol and provides an `exportedObject` that implements `receiveAXEvent`. Whichever side receives an event uses the sibling proxy to call into the other.

Since Zig can't directly conform to an @objc protocol, the Zig side needs a tiny Swift or Obj-C shim that: (a) maintains the NSXPCConnection, (b) exposes `receiveAXEvent` to Swift and forwards to a Zig C callback. Plan: ship a tiny `helper/Sources/ShokiXPCClient/ShokiXPCClient.swift` library target with a C-callable API (`@_cdecl`) that Zig links against, same build unit as the helper.
</xpc_callback_strategy>

       ```swift
       import Foundation
       import ApplicationServices
       import CoreFoundation

       public final class AXObserverSession {
           public typealias EventCallback = (_ phrase: String, _ tsNanos: UInt64, _ role: String?, _ name: String?) -> Void

           private var observer: AXObserver?
           private var runLoopSource: CFRunLoopSource?
           private var runLoop: CFRunLoop?
           private var observerThread: Thread?
           private let callback: EventCallback
           private var started = false
           private let lock = NSLock()

           public init(callback: @escaping EventCallback) {
               self.callback = callback
           }

           public func start(voicePID: pid_t) throws {
               lock.lock(); defer { lock.unlock() }
               if started { return }

               // Create observer targeting the VoiceOver process
               var obs: AXObserver?
               let selfPtr = Unmanaged.passUnretained(self).toOpaque()
               let err = AXObserverCreate(voicePID, AXObserverSession.axCCallback, &obs)
               guard err == .success, let observer = obs else {
                   throw NSError(domain: "org.shoki.runner", code: Int(err.rawValue),
                                 userInfo: [NSLocalizedDescriptionKey: "AXObserverCreate failed with \(err.rawValue)"])
               }
               self.observer = observer

               let systemWide = AXUIElementCreateSystemWide()
               let addErr = AXObserverAddNotification(observer, systemWide,
                   kAXAnnouncementRequestedNotification as CFString, selfPtr)
               guard addErr == .success else {
                   throw NSError(domain: "org.shoki.runner", code: Int(addErr.rawValue),
                                 userInfo: [NSLocalizedDescriptionKey: "AXObserverAddNotification failed with \(addErr.rawValue)"])
               }

               // Run the observer on a dedicated thread with its own runloop
               let thread = Thread { [weak self] in
                   guard let self = self, let obs = self.observer else { return }
                   let rl = CFRunLoopGetCurrent()
                   self.runLoop = rl
                   let src = AXObserverGetRunLoopSource(obs)
                   self.runLoopSource = src
                   CFRunLoopAddSource(rl, src, .defaultMode)
                   CFRunLoopRun()
               }
               thread.name = "shoki-ax-observer"
               thread.start()
               self.observerThread = thread
               started = true
           }

           public func stop() {
               lock.lock(); defer { lock.unlock() }
               if !started { return }
               if let rl = runLoop, let src = runLoopSource {
                   CFRunLoopRemoveSource(rl, src, .defaultMode)
                   CFRunLoopStop(rl)
               }
               runLoopSource = nil
               runLoop = nil
               observer = nil
               observerThread = nil
               started = false
           }

           /// Test-only: manually trigger an event as if the AX callback fired.
           public func debugEmit(phrase: String, tsNanos: UInt64, role: String?, name: String?) {
               callback(phrase, tsNanos, role, name)
           }

           private static let axCCallback: AXObserverCallbackWithInfo = { _, _, notification, userInfo, refcon in
               guard let refcon = refcon else { return }
               let session = Unmanaged<AXObserverSession>.fromOpaque(refcon).takeUnretainedValue()
               let dict = userInfo as? [String: Any] ?? [:]
               let phrase = (dict["AXAnnouncement"] as? String) ?? ""
               let name = dict["AXUIElementTitle"] as? String
               let role: String? = nil  // VO notifications typically don't carry role; caller can enrich later
               let ts = UInt64(Date().timeIntervalSince1970 * 1_000_000_000)
               session.callback(phrase, ts, role, name)
           }
       }
       ```
       ⚠ API hazard: `AXObserverCallbackWithInfo` vs `AXObserverCallback` — the "WithInfo" variant passes the userInfo dict and is what we want. Use `AXObserverCreateWithInfoCallback(pid, axCCallback, &obs)` if the basic `AXObserverCreate` signature mismatches on Swift 5.9+. Refer to `<ApplicationServices/HIServices/AXNotificationConstants.h>` headers — if Swift bridging prefers the WithInfo variant, use it. If both compile, prefer WithInfo.

       ```swift
       import Foundation
       import ShokiRunnerProtocol

       // C-callable API for Zig. Each @_cdecl exports an underscore-prefixed symbol matching the fn name.

       /// Opaque handle (actually a retained Swift class).
       private final class XPCClientSession: NSObject, ShokiClientProtocol {
           var connection: NSXPCConnection?
           var callback: (@convention(c) (UnsafePointer<Int8>, UInt64, UnsafePointer<Int8>?, UnsafePointer<Int8>?, UnsafeMutableRawPointer?) -> Void)? = nil
           var userdata: UnsafeMutableRawPointer? = nil

           func receiveAXEvent(phrase: String, tsNanos: UInt64, role: String?, name: String?) {
               guard let cb = callback else { return }
               phrase.withCString { phrasePtr in
                   let rolePtr: UnsafePointer<Int8>? = role?.cString(using: .utf8)?.withUnsafeBufferPointer { $0.baseAddress }
                   let namePtr: UnsafePointer<Int8>? = name?.cString(using: .utf8)?.withUnsafeBufferPointer { $0.baseAddress }
                   cb(phrasePtr, tsNanos, rolePtr, namePtr, userdata)
               }
           }
       }

       @_cdecl("shoki_xpc_connect")
       public func shoki_xpc_connect() -> UnsafeMutableRawPointer? {
           let session = XPCClientSession()
           let conn = NSXPCConnection(machServiceName: ShokiRunnerMachServiceName, options: [])
           conn.remoteObjectInterface = NSXPCInterface(with: ShokiRunnerProtocol.self)
           conn.exportedInterface = NSXPCInterface(with: ShokiClientProtocol.self)
           conn.exportedObject = session
           conn.resume()
           session.connection = conn
           return Unmanaged.passRetained(session).toOpaque()
       }

       @_cdecl("shoki_xpc_set_event_callback")
       public func shoki_xpc_set_event_callback(_ handle: UnsafeMutableRawPointer,
                                                _ cb: @escaping @convention(c) (UnsafePointer<Int8>, UInt64, UnsafePointer<Int8>?, UnsafePointer<Int8>?, UnsafeMutableRawPointer?) -> Void,
                                                _ userdata: UnsafeMutableRawPointer?) {
           let session = Unmanaged<XPCClientSession>.fromOpaque(handle).takeUnretainedValue()
           session.callback = cb
           session.userdata = userdata
       }

       @_cdecl("shoki_xpc_start_ax_observer")
       public func shoki_xpc_start_ax_observer(_ handle: UnsafeMutableRawPointer, _ pid: Int32) -> Int32 {
           let session = Unmanaged<XPCClientSession>.fromOpaque(handle).takeUnretainedValue()
           guard let proxy = session.connection?.remoteObjectProxy as? ShokiRunnerProtocol else { return -1 }
           let sem = DispatchSemaphore(value: 0)
           var rc: Int32 = 0
           proxy.startAXObserver(voicePID: pid) { err in
               if let _ = err { rc = -2 }
               sem.signal()
           }
           _ = sem.wait(timeout: .now() + 5)
           return rc
       }

       @_cdecl("shoki_xpc_stop_ax_observer")
       public func shoki_xpc_stop_ax_observer(_ handle: UnsafeMutableRawPointer) -> Int32 {
           let session = Unmanaged<XPCClientSession>.fromOpaque(handle).takeUnretainedValue()
           guard let proxy = session.connection?.remoteObjectProxy as? ShokiRunnerProtocol else { return -1 }
           let sem = DispatchSemaphore(value: 0)
           proxy.stopAXObserver { _ in sem.signal() }
           _ = sem.wait(timeout: .now() + 2)
           return 0
       }

       @_cdecl("shoki_xpc_disconnect")
       public func shoki_xpc_disconnect(_ handle: UnsafeMutableRawPointer) {
           let session = Unmanaged<XPCClientSession>.fromOpaque(handle).takeRetainedValue()
           session.connection?.invalidate()
           session.connection = nil
       ```
