Update `helper/scripts/build-app-bundle.sh` to verify BOTH bundles after the single `zig build`:

```bash
#!/usr/bin/env bash
set -euo pipefail
CONFIG="release"
while [[ $# -gt 0 ]]; do
    case "$1" in
        --configuration) CONFIG="$2"; shift 2 ;;
        *) echo "Unknown arg: $1" >&2; exit 2 ;;
    esac
done

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
HELPER_DIR="$(cd "$SCRIPT_DIR/.." && pwd)"
cd "$HELPER_DIR"

OPT_FLAG="-Doptimize=ReleaseFast"
[[ "$CONFIG" == "debug" ]] && OPT_FLAG="-Doptimize=Debug"

echo "[build-app-bundle] Building Zig helper (configuration=$CONFIG)"
zig build "$OPT_FLAG"

RUNNER="$HELPER_DIR/.build/ShokiRunner.app"
SETUP="$HELPER_DIR/.build/ShokiSetup.app"
DYLIB="$HELPER_DIR/.build/libShokiXPCClient.dylib"

for f in "$RUNNER/Contents/MacOS/ShokiRunner" \
         "$RUNNER/Contents/Info.plist" \
         "$SETUP/Contents/MacOS/ShokiSetup" \
         "$SETUP/Contents/Info.plist" \
         "$DYLIB"; do
    [[ -e "$f" ]] || { echo "error: missing artifact $f" >&2; exit 1; }
done

echo "[build-app-bundle] Bundles ready:"
echo "  - $RUNNER"
echo "  - $SETUP"
echo "  - $DYLIB"
```

Update `helper/scripts/sign-and-notarize.sh` to accept `[--entitlements <path>]` with a default:

```bash
#!/usr/bin/env bash
set -euo pipefail

APP_PATH=""
ENT_PATH=""
while [[ $# -gt 0 ]]; do
    case "$1" in
        --entitlements) ENT_PATH="$2"; shift 2 ;;
        *) if [[ -z "$APP_PATH" ]]; then APP_PATH="$1"; shift; else echo "Unknown arg: $1" >&2; exit 2; fi ;;
    esac
done

[[ -n "$APP_PATH" ]] || { echo "usage: $0 <path-to-app> [--entitlements <path>]" >&2; exit 2; }
[[ -d "$APP_PATH" ]] || { echo "error: app bundle not found: $APP_PATH" >&2; exit 1; }

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
HELPER_DIR="$(cd "$SCRIPT_DIR/.." && pwd)"

# Default: infer from bundle basename. ShokiRunner.app uses runner entitlements;
# ShokiSetup.app uses setup entitlements. Both are identical (Plan 03) but the
# path-per-bundle keeps semantic clarity.
if [[ -z "$ENT_PATH" ]]; then
    base="$(basename "$APP_PATH" .app)"
    case "$base" in
        ShokiRunner) ENT_PATH="$HELPER_DIR/src/runner/ShokiRunner.entitlements" ;;
        ShokiSetup)  ENT_PATH="$HELPER_DIR/src/setup/ShokiSetup.entitlements" ;;
        *) ENT_PATH="$HELPER_DIR/src/runner/ShokiRunner.entitlements" ;;
    esac
fi
[[ -f "$ENT_PATH" ]] || { echo "error: entitlements not found: $ENT_PATH" >&2; exit 1; }

: "${APPLE_DEVELOPER_ID_APP:?APPLE_DEVELOPER_ID_APP must be set}"
: "${APPLE_ID:?APPLE_ID must be set}"
: "${APPLE_TEAM_ID:?APPLE_TEAM_ID must be set}"
: "${APPLE_APP_SPECIFIC_PASSWORD:?APPLE_APP_SPECIFIC_PASSWORD must be set}"

echo "[sign] Codesigning $APP_PATH with entitlements $ENT_PATH"
codesign --force --options runtime --entitlements "$ENT_PATH" \
    --sign "$APPLE_DEVELOPER_ID_APP" --timestamp --deep "$APP_PATH"

codesign --verify --verbose=4 "$APP_PATH"
codesign -dvvv "$APP_PATH" 2>&1 | tee "$HELPER_DIR/.codesign.$(basename "$APP_PATH").out"

ZIP_PATH="$HELPER_DIR/$(basename "$APP_PATH").zip"
/usr/bin/ditto -c -k --keepParent "$APP_PATH" "$ZIP_PATH"

xcrun notarytool submit "$ZIP_PATH" \
    --apple-id "$APPLE_ID" --team-id "$APPLE_TEAM_ID" \
    --password "$APPLE_APP_SPECIFIC_PASSWORD" --wait --output-format json \
    | tee "$HELPER_DIR/.notarize.$(basename "$APP_PATH").json"

STATUS=$(node -e "const d=require('$HELPER_DIR/.notarize.$(basename "$APP_PATH").json'); console.log(d.status || 'unknown')")
[[ "$STATUS" == "Accepted" ]] || { echo "error: notarization failed with status $STATUS" >&2; exit 1; }

xcrun stapler staple "$APP_PATH"
xcrun stapler validate "$APP_PATH"

rm -f "$ZIP_PATH"
echo "[done] Signed + notarized: $APP_PATH"
```

```yaml
name: Build helper apps
description: >-
  Builds ShokiRunner.app AND ShokiSetup.app via helper/scripts/build-app-bundle.sh
  (zig build) and optionally signs + notarizes both when Apple secrets are present.
  Drops both built .app bundles into the binding package's helper/ directory.

inputs:
  binding-package-dir:
    description: "Path to the @shoki/binding-<os>-<arch> package directory"
    required: true
  sign:
    description: "Sign + notarize the built .apps (requires APPLE_* env)"
    required: false
    default: "false"

runs:
  using: composite
  steps:
    - name: Install Zig 0.16
      shell: bash
      run: |
        if ! command -v zig >/dev/null || [[ "$(zig version)" != 0.16.* ]]; then
            brew install zig
        fi
        zig version

    - name: Build both app bundles
      shell: bash
      working-directory: helper
      run: |
        set -euo pipefail
        ./scripts/build-app-bundle.sh
        ls -la .build/ShokiRunner.app .build/ShokiSetup.app

    - name: Sign + notarize both bundles
      if: inputs.sign == 'true'
      shell: bash
      working-directory: helper
      run: |
        set -euo pipefail
        if [[ -z "${DEVELOPER_ID_IDENTITY:-}" && -z "${APPLE_DEVELOPER_ID_APP:-}" ]]; then
            echo "warn: no sign identity set — skipping sign+notarize"
            exit 0
        fi
        export APPLE_DEVELOPER_ID_APP="${APPLE_DEVELOPER_ID_APP:-$DEVELOPER_ID_IDENTITY}"
        for app in .build/ShokiRunner.app .build/ShokiSetup.app; do
            ./scripts/sign-and-notarize.sh "$app"
        done

    - name: Copy signed bundles into binding package
      shell: bash
      run: |
        set -euo pipefail
        TARGET_DIR="$GITHUB_WORKSPACE/${{ inputs.binding-package-dir }}/helper"
        mkdir -p "$TARGET_DIR"
        cp -R helper/.build/ShokiRunner.app "$TARGET_DIR/"
        cp -R helper/.build/ShokiSetup.app "$TARGET_DIR/"
        ls -la "$TARGET_DIR/"
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CI env vars (APPLE_*) -> helper scripts | Secrets flow from GitHub Secrets → env → shell. A leak in `set -x` or an `echo $APPLE_APP_SPECIFIC_PASSWORD` would expose credentials in public CI logs. |
| build-helper-app composite action -> user repositories (future) | This composite action is used only inside this repo, not published to the marketplace. |
| CI smoke launching ShokiSetup in headless mode | `--self-test` must NOT show a UI modal or trigger a TCC prompt; if it does, CI hangs. Plan 03's self-test path enforces this. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-15 | I (Info Disclosure) | APPLE_* env vars in CI logs | mitigate | Scripts use `set -euo pipefail`, not `set -x`. Secrets come from `secrets.*` in GH Actions (redacted in logs automatically). No `echo $APPLE_*` anywhere in the updated scripts. |
| T-08-16 | R (Repudiation) | Notarization silently fails | mitigate | Script parses notarytool JSON; `[[ "$STATUS" == "Accepted" ]] || exit 1` gates success. Per-bundle `.notarize.<name>.json` files kept in helper/ (gitignored). |
| T-08-17 | D (DoS) | ShokiSetup blocks CI when self-test misses | mitigate | helper-smoke job has implicit 5-minute job timeout; self-test is designed to exit in <2s. If a regression makes --self-test hang, CI fails rather than leaking a runaway runner. |
| T-08-18 | T (Tampering) | build-helper-app copies wrong files | mitigate | New `ls -la` verify step after copy lists both .app bundles. Release verify step codesign -dvvv's both. |
| T-08-19 | S (Spoofing) | Third-party action pinned by major tag | accept | Using apple-actions/import-codesign-certs@v3 (same as pre-Plan-08). Not a new exposure. |
</threat_model>
