```zig
const std = @import("std");
const ax = @import("ax_bindings.zig");

pub const EventCallback = *const fn (phrase: []const u8, ts_nanos: u64, role: ?[]const u8, name: ?[]const u8, ctx: *anyopaque) void;

pub const AxError = error{
    InvalidPid,
    AlreadyStarted,
    CreateFailed,
    AddNotificationFailed,
};

pub const Session = struct {
    callback: EventCallback,
    ctx: *anyopaque,
    observer: ?ax.AXObserverRef = null,
    target_element: ?ax.AXUIElementRef = null,
    run_loop: ?ax.CFRunLoopRef = null,
    run_loop_source: ?ax.CFRunLoopSourceRef = null,
    started: bool = false,
    mu: std.Thread.Mutex = .{},
    observer_thread: ?std.Thread = null,

    pub fn init(callback: EventCallback, ctx: *anyopaque) Session { ... }

    pub fn start(self: *Session, target_pid: ax.pid_t) AxError!void {
        if (target_pid <= 0) return AxError.InvalidPid;
        self.mu.lock();
        defer self.mu.unlock();
        if (self.started) return; // idempotent per AXObserver.swift:58
        // ... AXObserverCreateWithInfoCallback, AddNotification, spawn thread, CFRunLoopRun
        self.started = true;
    }

    pub fn stop(self: *Session) void {
        self.mu.lock();
        defer self.mu.unlock();
        if (!self.started) return; // no-op per AXObserver.swift:111
        // ... CFRunLoopRemoveSource, CFRunLoopStop, join thread, CFRelease
        self.started = false;
    }

    pub fn debugEmit(self: *Session, phrase: []const u8, ts_nanos: u64, role: ?[]const u8, name: ?[]const u8) void {
        self.callback(phrase, ts_nanos, role, name, self.ctx);
    }

    pub fn isStarted(self: *Session) bool { ... }

    // Internal C-callable callback that matches AXObserverCallbackWithInfo signature.
    fn axCCallback(
        _: ax.AXObserverRef,
        _: ax.AXUIElementRef,
        _: ax.CFStringRef,
        userInfo: ax.CFDictionaryRef,
        refcon: ?*anyopaque,
    ) callconv(.c) void { ... }
};
```
