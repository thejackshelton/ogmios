```zig
const std = @import("std");

pub fn build(b: *std.Build) void {
    const target = b.standardTargetOptions(.{});
    const optimize = b.standardOptimizeOption(.{});

    // ShokiRunner XPC service core (static library in Plan 01 — Plan 02 promotes
    // this to link into the ShokiRunner executable). Keeping it static in Plan 01
    // lets the unit test binary link against the same compile unit.
    const xpc_core_mod = b.createModule(.{
        .root_source_file = b.path("src/runner/xpc_service.zig"),
        .target = target,
        .optimize = optimize,
    });

    const xpc_core = b.addLibrary(.{
        .name = "shoki_xpc_core",
        .linkage = .static,
        .root_module = xpc_core_mod,
    });
    // Linking against libxpc is implicit on macOS — no linkSystemLibrary call.
    if (target.result.os.tag == .macos) {
        xpc_core_mod.linkFramework("Foundation", .{});
    }
    b.installArtifact(xpc_core);

    // Unit tests.
    const test_step = b.step("test", "Run helper unit tests");

    const test_mod = b.createModule(.{
        .root_source_file = b.path("test/xpc_service_test.zig"),
        .target = target,
        .optimize = optimize,
    });
    if (target.result.os.tag == .macos) {
        test_mod.linkFramework("Foundation", .{});
    }
    const t = b.addTest(.{ .root_module = test_mod });
    const run_t = b.addRunArtifact(t);
    test_step.dependOn(&run_t.step);
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Zig test binary -> Zig production code | Tests use `dispatchForTest`; production uses `dispatch` against real xpc_object_t. A test-only stub leaking into a release build would allow a spoofed response. |
| XPC bindings -> `<xpc/xpc.h>` C ABI | Any ABI drift in the extern decls produces silent memory corruption at runtime. Hand-written decls must match Apple's header EXACTLY. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-01 | T (Tampering) | xpc_bindings.zig extern decls | mitigate | Hand-type decls from Apple's public `<xpc/xpc.h>`; add a trailing comment forbidding `@cImport`. Plan 02 adds a runtime smoke check that calls a known XPC function and confirms return values. |
| T-08-02 | I (Info Disclosure) | dispatchForTest leaking into production | mitigate | `dispatchForTest` is documented as test-only in its doc comment; real XPC listener in Plan 02 wires `dispatch` (not `dispatchForTest`). Plan 02's executable target explicitly calls only `dispatch`. |
| T-08-03 | E (Elevation) | Unknown-method arg → crash | mitigate | Dispatcher branches on `std.mem.eql`; falls through to a safe `unknown_method` errorCode response. Unit test 5 above proves the no-crash behavior. |
| T-08-04 | D (DoS) | startAXObserver with voicePID=0 or negative | mitigate | Validation mirrors ShokiRunnerService.swift line 56 (voicePID > 0); unit test 2 proves rejection. |
</threat_model>
