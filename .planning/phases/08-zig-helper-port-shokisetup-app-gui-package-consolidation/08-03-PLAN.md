<flow_contract>
<!-- The setup app's run-through. Implementors must preserve this order. -->

1. Parse argv:
   --version       -> print "ShokiSetup <ver> (zig-compiled)"; exit 0
   --self-test     -> run through init/alloc paths; skip modal; exit 0
   (no flag)       -> interactive run (below)

2. Interactive run:
   a. Call NSApplication.sharedApplication + setActivationPolicy(NSApplicationActivationPolicyRegular=0) + activateIgnoringOtherApps(YES)
   b. Probe Accessibility:
      - Build CFDictionary { kAXTrustedCheckOptionPrompt -> kCFBooleanTrue }
      - trusted = AXIsProcessTrustedWithOptions(opts)
      - If !trusted: macOS has already shown the system Accessibility prompt; wait via NSAlert saying "Click OK after granting Accessibility access" (manual wait — no busy loop).
   c. Probe Automation for VoiceOver:
      - Launch VO: NSAppleScript with source 'tell application "VoiceOver" to launch' — executeAndReturnError
      - Sleep 2s (or NSThread sleepForTimeInterval) for VO to finish boot
      - Send a real AppleEvent: NSAppleScript with 'tell application "VoiceOver" to get bounds of vo cursor' — executeAndReturnError
        This is the trigger per QA-REPORT.md Unblockers Step 1 Option A.
      - Regardless of result: the Automation TCC prompt has been displayed (or silently denied).
   d. Success modal: NSAlert "Shoki is ready. You can close this window." + addButtonWithTitle "Close" + runModal
   e. Cleanup: NSAppleScript 'tell application "VoiceOver" to quit' (courtesy; mirrors QA-REPORT.md Reproduce section)
   f. [NSApp terminate:nil]

3. Self-test (--self-test):
   Same init sequence up to step 2a, but skip steps 2b-e (no modal, no AppleScript), then terminate.
   Purpose: CI can exercise the linker + AppKit resolution on every build without blocking on user input.
</flow_contract>

const std = @import("std");
const ak = @import("appkit_bindings.zig");

pub const version = "0.1.0";

pub const CmdMode = enum { interactive, version, self_test };

pub const ParseError = error{ UnknownFlag };

pub fn parseArgs(arg: []const u8) ParseError!CmdMode {
    if (arg.len == 0) return .interactive;
    if (std.mem.eql(u8, arg, "--version")) return .version;
    if (std.mem.eql(u8, arg, "--self-test")) return .self_test;
    return ParseError.UnknownFlag;
}

pub fn main() !void {
    var args = std.process.args();
    _ = args.next();
    const first = args.next() orelse "";
    const mode = parseArgs(first) catch {
        const stderr = std.io.getStdErr().writer();
        try stderr.print("ShokiSetup: unknown flag '{s}'\n", .{first});
        std.process.exit(2);
    };

    switch (mode) {
        .version => {
            const out = std.io.getStdOut().writer();
            try out.print("ShokiSetup {s} (zig-compiled)\n", .{version});
            return;
        },
        .self_test => {
            // Wire AppKit to prove linkage, but DON'T show any UI or send AppleEvents.
            _ = ak.objc_getClass("NSObject") orelse {
                std.process.exit(10); // Obj-C runtime missing
            };
            _ = ak.objc_getClass("NSApplication") orelse {
                std.process.exit(11); // AppKit not linked
            };
            _ = ak.objc_getClass("NSAppleScript") orelse {
                std.process.exit(12); // Foundation incomplete
            };
            // Success: exit 0 without showing UI.
            return;
        },
        .interactive => try runInteractive(),
    }
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| ShokiSetup.app -> macOS TCC subsystem | Info.plist CFBundleIdentifier = org.shoki.setup. Grants persist per-bundle-ID. If the plist is wrong, grants go to the wrong identity. |
| setup_main.zig -> AppKit/Obj-C runtime | objc_msgSend is untyped at the ABI. Wrong selector/signature = undefined behavior at runtime. |
| Obj-C runtime -> VoiceOver via AppleScript | VO receives a "get bounds of vo cursor" event. If VO interprets this wrongly (e.g. modifies cursor state), it's user-visible. The event is read-only per Apple docs so this is accept. |
