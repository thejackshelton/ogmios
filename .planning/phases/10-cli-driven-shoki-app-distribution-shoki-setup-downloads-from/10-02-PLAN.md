<objective>
Replace the existing `shoki setup` body (which only resolves a path inside `node_modules/@shoki/binding-*/helper/`) with a full download-and-install flow: detect install state in `~/Applications/`, fetch the unified `shoki-<platform>.zip` from GitHub Releases, verify its `.sha256`, unzip into `~/Applications/`, strip quarantine, then launch `Shoki Setup.app`.

Purpose: Phase 10's headline DX is `npm install shoki && npx shoki setup` — this plan ships the second half. After this plan runs, no consumer ever needs to visit GitHub Releases manually or copy `.app` bundles around.

Output: Working `shoki setup` that, on a clean machine, ends with both `~/Applications/Shoki.app` and `~/Applications/Shoki Setup.app` present, quarantine-stripped, and `Shoki Setup.app` opened. All flags from CONTEXT.md wired.
</objective>

```typescript
// packages/sdk/src/cli/setup-download.ts
export type ShokiAppPlatform = 'darwin-arm64' | 'darwin-x64';

export interface DownloadOptions {
  /** GitHub release tag, e.g. "app-v0.1.0". Default: read compatibleAppVersion from sdk package.json. */
  version: string;
  /** Override base URL — used in tests to point at a fixture server. Default: github.com release URL. */
  baseUrl?: string;
  /** Resolved platform key. */
  platform: ShokiAppPlatform;
  /** Where to write the downloaded zip + sha file. Default: a tmp dir under os.tmpdir(). */
  workDir?: string;
  /** Injected fetch — defaults to global fetch. Used in tests. */
  fetch?: typeof globalThis.fetch;
  /** Injected fs — defaults to node:fs/promises. Used in tests. */
  fs?: { writeFile: typeof import('node:fs/promises').writeFile; mkdtemp: typeof import('node:fs/promises').mkdtemp };
}

export interface DownloadResult {
  zipPath: string;
  shaPath: string;
  expectedSha: string;
  actualSha: string;
  bytesDownloaded: number;
}

export async function downloadAndVerifyZip(opts: DownloadOptions): Promise<DownloadResult>;

// packages/sdk/src/cli/setup-install.ts
export interface InstallOptions {
  zipPath: string;
  installDir: string; // e.g. /Users/x/Applications
  /** Injected exec — defaults to execa. Used in tests. */
  exec?: (file: string, args: string[]) => Promise<{ stdout: string; stderr: string; exitCode: number }>;
}

export interface InstallResult {
  installedPaths: string[]; // ~/Applications/Shoki.app, ~/Applications/Shoki Setup.app
}

export async function installFromZip(opts: InstallOptions): Promise<InstallResult>;
export async function stripQuarantine(paths: string[], exec?: InstallOptions['exec']): Promise<void>;
export async function readInstalledAppVersion(appPath: string): Promise<string | null>;

// packages/sdk/src/cli/setup-command.ts
export interface SetupOptions {
  force?: boolean;          // --force
  noDownload?: boolean;     // --no-download
  installDir?: string;      // --install-dir <path>; defaults to ~/Applications
  skipLaunch?: boolean;     // --skip-launch
  json?: boolean;           // --json
  version?: string;         // --version <ver>; defaults to compatibleAppVersion
  dryRun?: boolean;         // dry-run (kept from existing impl)
}

export interface SetupResult {
  action: 'noop' | 'downloaded' | 'reinstalled' | 'launched-only';
  installDir: string;
  appPaths: string[];
  downloadedFromUrl?: string;
  shaVerified?: boolean;
  launched: boolean;
  exitCode: number;
}

export async function runSetup(opts: SetupOptions): Promise<SetupResult>;
```
