Purpose: Bridge browser-mode tests to Node-side `@shoki/sdk`. Browser-side tests never import Node-only modules (VITEST-02); they only call `commands.shokiStart(...)` which Vitest's tinyRPC routes over WebSocket to the Node handler. Every payload flowing over tinyRPC is structured-clone-safe: `tsNanos` is encoded as a number (milliseconds since epoch) rather than bigint, dates are ISO strings or numbers, no Functions (VITEST-06).

Output: A published-shape `packages/vitest/` with a Node plugin entry (`@shoki/vitest`), a browser entry (`@shoki/vitest/browser`), 10 command handler files, typed contracts in `command-types.ts`, an error taxonomy, and unit tests covering singleton detection + structured-clone safety of every return payload.

Scope boundary: This plan scaffolds the plugin surface and the browser proxy. The actual session lifecycle (start/stop refcounting, reset between tests, Node-side `@shoki/sdk` instantiation) lands in Plan 04-03. Command handlers in this plan return stubbed-but-structured payloads so tests can verify the tinyRPC wire shape; Plan 04-03 replaces the stub bodies with real `@shoki/sdk` calls.

    **`packages/vitest/src/browser.ts`** — browser-side session proxy. Uses `@vitest/browser/context`'s `commands` object; this file MUST be import-safe from a browser bundle (no `process.*`, no Node-only modules):

    ```ts
    import { commands } from '@vitest/browser/context';
    import { ShokiConcurrentTestError } from './errors.js';
    import type {
      ShokiStartArgs,
      ShokiStartResult,
      WireShokiEvent,
      ShokiAwaitStableArgs,
    } from './command-types.js';

    // Typed RPC. `commands` is `Record<string, (...args: unknown[]) => Promise<unknown>>` at Vitest's type level.
    // We re-type it against our command surface.
    interface ShokiRpc {
      shokiStart: (args: ShokiStartArgs) => Promise<ShokiStartResult>;
      shokiStop: (args: { sessionId: string }) => Promise<{ stopped: boolean; remainingRefs: number }>;
      shokiListen: (args: { sessionId: string; sinceMs?: number }) => Promise<WireShokiEvent[]>;
      shokiDrain: (args: { sessionId: string }) => Promise<WireShokiEvent[]>;
      shokiPhraseLog: (args: { sessionId: string }) => Promise<string[]>;
      shokiLastPhrase: (args: { sessionId: string }) => Promise<string | null>;
      shokiClear: (args: { sessionId: string }) => Promise<{ ok: true }>;
      shokiReset: (args: { sessionId: string }) => Promise<{ ok: true }>;
      shokiAwaitStable: (args: ShokiAwaitStableArgs) => Promise<WireShokiEvent[]>;
      shokiGetDroppedCount: (args: { sessionId: string }) => Promise<{ droppedCount: number }>;
    }
    const rpc = commands as unknown as ShokiRpc;

    export interface ShokiBrowserSession {
      readonly sessionId: string;
      stop(): Promise<{ stopped: boolean; remainingRefs: number }>;
      drain(): Promise<WireShokiEvent[]>;
      listen(sinceMs?: number): Promise<WireShokiEvent[]>;
      phraseLog(): Promise<string[]>;
      lastPhrase(): Promise<string | null>;
      clear(): Promise<void>;
      reset(): Promise<void>;
      awaitStable(opts: { quietMs: number; timeoutMs?: number }): Promise<WireShokiEvent[]>;
      droppedCount(): Promise<number>;
    }

    function detectConcurrentContext(): boolean {
      // Best-effort detection of `test.concurrent` — Vitest's `vi.getTestMeta()` exposes `concurrent`.
      // If `vi` is not available in browser context, return false.
      const g = globalThis as { vi?: { getTestMeta?: () => { concurrent?: boolean } | undefined } };
      try {
        const meta = g.vi?.getTestMeta?.();
        return meta?.concurrent === true;
      } catch {
        return false;
      }
    }

    export const voiceOver = {
      async start(args: ShokiStartArgs = {}): Promise<ShokiBrowserSession> {
        if (detectConcurrentContext()) {
          throw new ShokiConcurrentTestError();
        }
        const { sessionId } = await rpc.shokiStart(args);
        return {
          sessionId,
          stop: () => rpc.shokiStop({ sessionId }),
          drain: () => rpc.shokiDrain({ sessionId }),
          listen: (sinceMs) => rpc.shokiListen({ sessionId, sinceMs }),
          phraseLog: () => rpc.shokiPhraseLog({ sessionId }),
          lastPhrase: () => rpc.shokiLastPhrase({ sessionId }),
          clear: async () => { await rpc.shokiClear({ sessionId }); },
          reset: async () => { await rpc.shokiReset({ sessionId }); },
          awaitStable: (opts) => rpc.shokiAwaitStable({ sessionId, ...opts }),
          droppedCount: async () => (await rpc.shokiGetDroppedCount({ sessionId })).droppedCount,
        };
      },
    ```

    ```ts
    import { readdir, readFile, stat } from 'node:fs/promises';
    import { join, resolve } from 'node:path';

    const IMPORT_NEEDLE = '@shoki/vitest/browser';
    const TEST_EXTS = new Set(['.ts', '.tsx', '.js', '.jsx', '.mts', '.mjs']);

    export async function detectVoiceOverImports(rootDir: string): Promise<boolean> {
      async function walk(dir: string): Promise<boolean> {
        let entries;
        try { entries = await readdir(dir, { withFileTypes: true }); }
        catch { return false; }
        for (const e of entries) {
          if (e.name === 'node_modules' || e.name === '.git' || e.name === 'dist' || e.name.startsWith('.')) continue;
          const full = join(dir, e.name);
          if (e.isDirectory()) {
            if (await walk(full)) return true;
            continue;
          }
          const idx = e.name.lastIndexOf('.');
          const ext = idx < 0 ? '' : e.name.slice(idx);
          if (!TEST_EXTS.has(ext)) continue;
          try {
            const src = await readFile(full, 'utf8');
            if (src.includes(IMPORT_NEEDLE)) return true;
          } catch { /* ignore read errors */ }
        }
        return false;
      }
      return walk(resolve(rootDir));
    }
    ```
