## sessionId vs ScreenReaderHandle

Vitest process-local model:

```
SessionStore
 ├─ handle: ScreenReaderHandle | null       // one underlying VO session
 ├─ startRefs: number                        // how many active sessionIds
 ├─ sessions: Map<string, SessionState>      // per-sessionId state
 └─ counter: number                          // monotonic id generator

SessionState {
  sessionId: string
  lastDrainedEventIndex: number              // for listen({ sinceMs }) -- use event position, not wall clock
  active: boolean                            // set false on stop(sessionId)
}
```

**Invariants:**
- `startRefs === 0` ⇒ `handle === null` (or fully deinit'd).
- `startRefs > 0` ⇒ `handle !== null` and has been `start()`-ed.
- `store.start()` flow: if `startRefs === 0`, instantiate `voiceOver(opts)` + await `handle.start()`. Then ALWAYS increment `startRefs`, create a new sessionId+SessionState. Return sessionId.
- `store.stop(sessionId)`: mark session inactive; decrement `startRefs`. If `startRefs === 0`, `await handle.stop(); await handle.deinit(); handle = null`.
- `store.reset(sessionId)`: `await handle.reset()`. Resets the SHARED log. Sessions tracking `lastDrainedEventIndex` reset their local cursor to 0.

**Why lastDrainedEventIndex and not wall-clock `sinceMs`:** The browser entry's `listen({ sinceMs })` sends a ms timestamp but that's inherently racy across the tinyRPC boundary. The Node-side implementation resolves `sinceMs` by looking at the current log and emitting events with `tsMs >= sinceMs`. That's fine for drain semantics. For `listen`-style windowing per session we ALSO track a per-session drain cursor so `session.drain()` on the browser side returns NEW events only since its last drain. This matches the browser-side proxy's natural usage pattern (call `drain()`, assert, call `drain()` again after next interaction, assert on the delta).

## Mapping Phase 3 API to BrowserCommands

| BrowserCommand | @shoki/sdk call |
|----------------|------------------|
| shokiStart | `voiceOver(opts); await handle.start()` (once; refcount++ on subsequent calls) |
| shokiStop | refcount--; if 0: `await handle.stop(); await handle.deinit()` |
| shokiDrain | `const events = await handle.drain(); cursor.update(); return events.map(toWire)` |
| shokiListen({ sinceMs }) | drain to internal LogStore then filter `events.filter(e => e.tsNanos >= sinceMs*1e6n).map(toWire)` — OR per-session cursor delta |
| shokiPhraseLog | `await handle.phraseLog()` → string[] |
| shokiLastPhrase | `(await handle.lastPhrase()) ?? null` (convert undefined → null for clone safety) |
| shokiClear | `await handle.clear()` — TS-side only per CAP-11 |
| shokiReset | `await handle.reset()` |
| shokiAwaitStable({ quietMs, timeoutMs }) | `const snapshot = await handle.awaitStableLog({ quietMs, signal: timeoutMs ? AbortSignal.timeout(timeoutMs) : undefined }); return snapshot.map(toWire)` |
| shokiGetDroppedCount | `const n = await handle.droppedCount(); return { droppedCount: Number(n) }` |

    ```ts
    import { type ScreenReaderHandle, type ShokiEvent, voiceOver, type VoiceOverOptions } from '@shoki/sdk';
    import { ShokiSessionNotFoundError } from './errors.js';
    import type { WireShokiEvent } from './command-types.js';

    export interface ShokiSdkDriver {
      create(opts: VoiceOverOptions): ScreenReaderHandle;
    }

    export const realShokiSdkDriver: ShokiSdkDriver = {
      create: (opts) => voiceOver(opts),
    };

    interface SessionState {
      sessionId: string;
      active: boolean;
      cursor: number;  // index into the shared log; all events at index >= cursor are "new" to this session
    }

    /**
     * Refcounted, process-local Shoki session registry (VITEST-05).
     *
     * Model:
     * - ONE underlying ScreenReaderHandle serves many sessionIds.
     * - First start() boots VO; subsequent starts mint a new sessionId and increment the refcount.
     * - Last stop() tears the handle down (stop + deinit).
     * - reset(sessionId) calls handle.reset() AND rewinds this session's cursor to 0 (the log was
     *   cleared, so "new events since last cursor" starts fresh).
     * - Each session keeps its own cursor so two concurrent `session.drain()` callers don't steal
     *   each other's events.
     *
     * NOT safe for cross-process sharing — each Vitest process has its own SessionStore.
     */
    export class SessionStore {
      private handle: ScreenReaderHandle | null = null;
      private startRefs = 0;
      private counter = 0;
      private sessions = new Map<string, SessionState>();

      /** Total events ever pushed to the shared log. Used as the cursor high-water mark. */
      private sharedEventLog: ShokiEvent[] = [];

      async start(driver: ShokiSdkDriver, opts: VoiceOverOptions): Promise<string> {
        if (this.handle === null) {
          const h = driver.create(opts);
          try {
            await h.start();
          } catch (err) {
            // Do NOT increment refcount on failure.
            throw err;
          }
          this.handle = h;
        }
        this.startRefs += 1;
        this.counter += 1;
        const sessionId = `shoki-${this.counter}`;
        this.sessions.set(sessionId, { sessionId, active: true, cursor: this.sharedEventLog.length });
        return sessionId;
      }

      async stop(sessionId: string): Promise<{ stopped: boolean; remainingRefs: number }> {
        const s = this.getSession(sessionId);
        if (!s.active) return { stopped: false, remainingRefs: this.startRefs };
        s.active = false;
        this.startRefs -= 1;
        const stopped = this.startRefs === 0;
        if (stopped && this.handle !== null) {
          const h = this.handle;
          this.handle = null;
          try { await h.stop(); } finally { await h.deinit(); }
          this.sharedEventLog = [];
        }
        return { stopped, remainingRefs: this.startRefs };
      }

      async reset(sessionId: string): Promise<void> {
        this.getSession(sessionId);  // validate
        const h = this.requireHandle();
        await h.reset();
        this.sharedEventLog = [];
        for (const s of this.sessions.values()) {
          s.cursor = 0;
        }
      }

      async clear(sessionId: string): Promise<void> {
        this.getSession(sessionId);
        const h = this.requireHandle();
        await h.clear();
        this.sharedEventLog = [];
        for (const s of this.sessions.values()) {
          s.cursor = 0;
        }
      }

      async drain(sessionId: string): Promise<WireShokiEvent[]> {
        const s = this.getSession(sessionId);
        await this.pumpHandleIntoSharedLog();
        const slice = this.sharedEventLog.slice(s.cursor);
        s.cursor = this.sharedEventLog.length;
        return slice.map(toWireEvent);
      }

      async listen(sessionId: string, sinceMs?: number): Promise<WireShokiEvent[]> {
        // sinceMs-based variant: return events with tsMs >= sinceMs (absolute wall clock).
        // If sinceMs is undefined, fall through to cursor-based drain.
        if (sinceMs === undefined) return this.drain(sessionId);
        this.getSession(sessionId);
        await this.pumpHandleIntoSharedLog();
        const cutoffNs = BigInt(sinceMs) * 1_000_000n;
        return this.sharedEventLog.filter((e) => e.tsNanos >= cutoffNs).map(toWireEvent);
      }

      async phraseLog(sessionId: string): Promise<string[]> {
        this.getSession(sessionId);
        const h = this.requireHandle();
        return h.phraseLog();
      }

      async lastPhrase(sessionId: string): Promise<string | null> {
        this.getSession(sessionId);
        const h = this.requireHandle();
        return (await h.lastPhrase()) ?? null;
      }

      async awaitStable(
        sessionId: string,
        opts: { quietMs: number; timeoutMs?: number },
      ): Promise<WireShokiEvent[]> {
        const s = this.getSession(sessionId);
        const h = this.requireHandle();
        const signal = opts.timeoutMs !== undefined ? AbortSignal.timeout(opts.timeoutMs) : undefined;
        const snapshot = await h.awaitStableLog({ quietMs: opts.quietMs, signal });
        // Refresh shared log + advance session cursor so follow-up drain() returns nothing until new events arrive.
        this.sharedEventLog = snapshot.slice();
        s.cursor = this.sharedEventLog.length;
        return snapshot.map(toWireEvent);
      }

      async getDroppedCount(sessionId: string): Promise<{ droppedCount: number }> {
        this.getSession(sessionId);
        const h = this.requireHandle();
        const n = await h.droppedCount();
        // droppedCount is a bigint; structured-clone requires number. Practical values fit in safe-int.
        return { droppedCount: Number(n) };
      }

      /** Internal: drain the handle's accumulated events into our mirror log. */
      private async pumpHandleIntoSharedLog(): Promise<void> {
        const h = this.requireHandle();
        const events = await h.drain();
        this.sharedEventLog.push(...events);
      }

      private getSession(sessionId: string): SessionState {
        const s = this.sessions.get(sessionId);
        if (!s) throw new ShokiSessionNotFoundError(sessionId);
        return s;
      }

      private requireHandle(): ScreenReaderHandle {
        if (this.handle === null) throw new ShokiSessionNotFoundError('<handle-already-torn-down>');
        return this.handle;
      }

      /** Test-only introspection. */
      get _startRefs(): number { return this.startRefs; }
      get _handle(): ScreenReaderHandle | null { return this.handle; }
    }

    /**
     * Convert a ShokiEvent (with bigint tsNanos) into a structured-clone-safe WireShokiEvent.
     * tsMs is an integer millisecond count (bigint-division, floor toward zero).
     */
    export function toWireEvent(e: ShokiEvent): WireShokiEvent {
      const tsMs = Number(e.tsNanos / 1_000_000n);
      return {
        tsMs,
        source: e.source,
        flags: e.flags,
        phrase: e.phrase,
        role: e.role,
        name: e.name,
      };
    }
    ```
