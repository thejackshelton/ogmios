restore-vo-settings` as a user-facing escape hatch.

CONTEXT.md states: "Failure to restore = we break the user's Mac's
accessibility. This is already implemented in Phase 3 (snapshot →

      const override = std.posix.getenv("SHOKI_SNAPSHOT_PATH");
      const path = if (override) |p| try self.allocator.dupe(u8, p) else blk: {

  ```ts
  import { execa } from 'execa';
  import { readFile } from 'node:fs/promises';
  import { homedir } from 'node:os';
  import { join } from 'node:path';

  export const DEFAULT_SNAPSHOT_PATH = join(homedir(), '.shoki', 'vo-snapshot.plist');
  const STALE_THRESHOLD_SECONDS = 7 * 24 * 60 * 60;

  export interface RestoreResult {
    ok: boolean;
    code?: 'SNAPSHOT_MISSING' | 'SNAPSHOT_UNRECOGNIZED' | 'SNAPSHOT_STALE' | 'WRITE_FAILED' | 'OK';
    restoredKeys?: string[];
    failures?: Array<{ key: string; error: string }>;
    snapshotAgeSeconds?: number;
  }

  export interface RestoreOptions {
    snapshotPath?: string;
    force?: boolean;
  }

  // The 9 canonical keys shoki writes to com.apple.VoiceOver4; domain path
  // branches on macOS version but `defaults` handles domain resolution if
  // we pass the domain name directly.
  const SHOKI_KEYS_TYPED: Array<{ key: string; type: 'int' | 'real' | 'bool' }> = [
    { key: 'SCREnableAppleScript', type: 'int' },
    { key: 'SpeechRate', type: 'real' },
    { key: 'VoiceVolume', type: 'real' },
    { key: 'EffectsEnabled', type: 'int' },
    { key: 'HotSpotTracking', type: 'int' },
    { key: 'NumpadCommander', type: 'int' },
    { key: 'ElectronicSoundsEnabled', type: 'int' },
    { key: 'MagnifierEnabled', type: 'int' },
    { key: 'AutoVoiceCursor', type: 'int' },
  ];

  export async function restoreVoSettingsFromSnapshot(opts: RestoreOptions = {}): Promise<RestoreResult> {
    const path = opts.snapshotPath ?? DEFAULT_SNAPSHOT_PATH;
    let xml: string;
    try { xml = await readFile(path, 'utf8'); }
    catch { return { ok: false, code: 'SNAPSHOT_MISSING' }; }

    // Minimal plist XML parse — we're reading our own file, strict regex is fine.
    const versionMatch = xml.match(/<key>_shoki_snapshot_version<\/key>\s*<integer>(\d+)<\/integer>/);
    if (!versionMatch) return { ok: false, code: 'SNAPSHOT_UNRECOGNIZED' };

    const tsMatch = xml.match(/<key>_shoki_snapshot_ts_unix<\/key>\s*<integer>(\d+)<\/integer>/);
    const now = Math.floor(Date.now() / 1000);
    const ts = tsMatch ? parseInt(tsMatch[1], 10) : 0;
    const age = now - ts;
    if (age > STALE_THRESHOLD_SECONDS && !opts.force) {
      return { ok: false, code: 'SNAPSHOT_STALE', snapshotAgeSeconds: age };
    }

    const restored: string[] = [];
    const failures: Array<{ key: string; error: string }> = [];
    for (const { key, type } of SHOKI_KEYS_TYPED) {
      // Extract value from xml. Each key appears as <key>NAME</key><TYPE_TAG>VALUE</TYPE_TAG>.
      const tagName = type === 'int' ? 'integer' : type === 'real' ? 'real' : 'true|false';
      const re = new RegExp(`<key>${key}</key>\\s*<(${tagName})(?:[^>]*)>([^<]+)</\\1>|<key>${key}</key>\\s*<(true|false)\\s*/>`);
      const m = xml.match(re);
      if (!m) continue; // key absent from snapshot — skip
      const value = m[2] ?? m[3] ?? '';
      try {
        // defaults write com.apple.VoiceOver4 <key> -<type> <value>
        await execa('defaults', ['write', 'com.apple.VoiceOver4', key, `-${type}`, value]);
        restored.push(key);
      } catch (err) {
        failures.push({ key, error: (err as Error).message });
      }
    }
    if (failures.length > 0) return { ok: false, code: 'WRITE_FAILED', restoredKeys: restored, failures };
    return { ok: true, code: 'OK', restoredKeys: restored };
  }
  ```
